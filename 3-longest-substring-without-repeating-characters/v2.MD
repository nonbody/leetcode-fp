const isCharExist = (target: string, longest: string[]): boolean =>
  longest.includes(target);

const shiftOutUntilFound = (target: string, longest: string[]): string[] => {
  const shift = longest.shift();
  return target !== shift ? shiftOutUntilFound(target, longest) : longest;
};

const findLatestLongestLength = (
  target: string,
  longest: string[],
  max: number
): number => {
  let prevMax = max;
  if (isCharExist(target, longest)) {
    prevMax = Math.max(longest.length, max);
    longest = shiftOutUntilFound(target, longest);
  }
  longest.push(target);
  return Math.max(prevMax, longest.length);
};

const lengthOfLongestSubstring = (input: string): number => {
  const longest: string[] = [];
  let max = 0;

  for (let i = 0; i < input.length; i++) {
    max = findLatestLongestLength(input[i], longest, max);
  }

  return max;
};
