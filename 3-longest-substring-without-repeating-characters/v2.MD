const isCharExist = (target: string, longest: string[]): boolean =>
  longest.includes(target);

const shiftOutTilFound = (target: string, longest: string[]): string[] => {
  const shift = longest.shift();
  return target !== shift ? shiftOutTilFound(target, longest) : longest;
};

const findLatestLongestLength = (
  target: string,
  longest: string[],
  max: number
): number => {
  let prevMax = max;
  if (isCharExist(target, longest)) {
    prevMax = Math.max(longest.length, max);
    longest = shiftOutTilFound(target, longest);
  }
  longest.push(target);
  return Math.max(longest.length, prevMax);
};

const lengthOfLongestSubstring = (input: string): number => {
  const longest: string[] = [];
  let max = 0;

  for (let i = 0; i < input.length; i++) {
    max = findLatestLongestLength(input.charAt(i), longest, max);
  }

  return max;
};
